<script>
let recorder = null;
let chunks = [];
let stream = null;

const startBtn = document.getElementById("start");
const stopBtn = document.getElementById("stop");
const statusText = document.getElementById("status");
const resultBox = document.getElementById("result");
const csrfToken = document.querySelector("meta[name='csrf-token']").content;

startBtn.onclick = async function(){

    try{

        stream = await navigator.mediaDevices.getUserMedia({ audio:true });

        recorder = new MediaRecorder(stream);
        chunks = [];

        recorder.ondataavailable = function(e){
            chunks.push(e.data);
        };

        recorder.onstop = async function(){

            statusText.innerText = "Processing...";

            const blob = new Blob(chunks, { type:"audio/webm" });

            const formData = new FormData();
            formData.append("audio", blob, "voice.webm");

            try{
                const response = await fetch("/analyze_voice", {
                    method:"POST",
                    headers:{
                        "X-CSRFToken": csrfToken
                    },
                    body: formData
                });

                const data = await response.json();

                if(data.error){
                    resultBox.innerHTML = `<p>⚠ Error: ${data.error}</p>`;
                }else{
                    resultBox.innerHTML = `
                        <h3>🎤 Voice Analysis</h3>
                        <p><b>${data.verdict || "Voice analyzed"}</b></p>
                        <p>📊 AI Confidence: ${data.ai_confidence}%</p>
                        <p>⏱ Duration: ${data.duration || "unknown"}s</p>
                    `;
                }

            }catch(err){
                resultBox.innerHTML = `<p>⚠ Upload failed: ${err.message}</p>`;
            }

            if(stream){
                stream.getTracks().forEach(track => track.stop());
            }

            startBtn.disabled = false;
            stopBtn.disabled = true;
            statusText.innerText = "Not recording";
        };

        recorder.start();

        startBtn.disabled = true;
        stopBtn.disabled = false;
        statusText.innerText = "🎤 Recording...";

    }catch(err){
        resultBox.innerHTML = `<p>⚠ Microphone error: ${err.message}</p>`;
    }
};

stopBtn.onclick = function(){

    if(recorder && recorder.state === "recording"){
        recorder.stop();
        statusText.innerText = "Stopping...";
    }

};
</script>
